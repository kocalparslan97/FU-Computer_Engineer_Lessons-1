package yigitdizi;
public class yigitDizi {
    int dizi[];
    int ElemanSayisi;
    static int r1,r2;
    int goster;
    
    yigitDizi(int ElemanSayisi){
        this.ElemanSayisi=ElemanSayisi;
        this.dizi=new int[ElemanSayisi];
        this.r1=-1;
        this.r2=ElemanSayisi-1;
    }
    
//EKLEME//
    void y1Push(int a){
        if(r1!=r2){
            r1++;
            dizi[r1]=a;
        }
        else{
            System.out.println("Yýðýt1 dolu!");
        }
    }
    
    void y2Push(int a){
        if(r1!=r2){
            dizi[r2]=a;
            r2--;
        }
        else{
            System.out.println("Yýðýt2 dolu!");
        }
    }

//SÝLME//
    void y1Pop() {
        if (r1 < 0) {
            System.out.println("Yýðýt1 Zaten Boþ");
        } else {
            System.out.println("Pop:|"+dizi[r1--]+"|");
        }
    }

    void y2Pop() {
        if (r2 > ElemanSayisi - 2) {
            System.out.println("Yýðýt2 Zaten Boþ");
        } else {
            System.out.println("Pop:|"+dizi[++r2]+"|");
        }
    }
    
//YAZDIRMA//
    void y1Goster(){
        
        for (int i = r1; i > -1; i--) {
            System.out.print("|"+dizi[i]+"|");
        }
    }
    
    void y2Goster(){
        int temp=r2+1;
        for (int i = ElemanSayisi-r2-1; i >0 ; i--) {
            System.out.print("|"+dizi[temp]+"|");
            
            temp++;
        }
    }
}
