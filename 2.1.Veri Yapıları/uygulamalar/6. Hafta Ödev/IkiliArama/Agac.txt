public class Agac {
    Yaprak root;
    Agac(){
        root=null;
    }
    
    void Ekle(Yaprak yeni){
        if(root==null){
            root=yeni;
        }
        else{
            
            Yaprak temp=root;
            boolean durum=true;
            while(durum){
                if(yeni.no < temp.no && temp.sol!=null){
                    temp=temp.sol;
                }
                else if(yeni.no < temp.no && temp.sol==null){
                    yeni.kok=temp;
                    yeni.kok.sol=yeni;
                    durum=false;
                }
                else if(yeni.no > temp.no && temp.sag!=null){
                    temp=temp.sag;
                }
                else if(yeni.no > temp.no && temp.sag==null){
                    yeni.kok=temp;
                    yeni.kok.sag=yeni;
                    durum=false;
                }  
            }
            
        }
    }//ekle() sonu
    
    void maxDegerBul(){
        Yaprak temp=root;
        while(temp.sag!=null){
            temp=temp.sag;
        }
        if(temp!=null){
            System.out.println("En buyuk deger="+temp.no);
        }
        else{
            System.out.println("En buyuk deger="+root.no);
        }
    }
    
    void minDegerBul(){
        Yaprak temp=root;
        while(temp.sol!=null){
            temp=temp.sol;
        }
        if(temp!=null){
            System.out.println("En kucuk deger="+temp.no);
        }
        else{
            System.out.println("En kucuk deger="+root.no);
        }
    }
    
    void arama(int no){
        Yaprak temp=root;
        boolean durum=true;
        boolean bulunma=false;
        while(durum){
            try{
                if(no==temp.no){
                    bulunma=true; durum=false;
                }
                else if(no>temp.no){
                    temp=temp.sag;
                }
                else if(no<temp.no){
                    temp=temp.sol;
                }
            }
            catch(Exception e){
                durum=false;
            }
            
            
        }
        if(bulunma==true){
            System.out.println(temp.no+" numarali ogrenci bulundu: "+temp.isim);
        }
        else{
            System.out.println("Ogrenci bulunamadi!");
        }
    }
    
    void silme(int no){
        Yaprak temp=root;
        Yaprak temponceki=null;
        Yaprak bos=null;
        boolean durum=true;
        boolean bulunma=false;
        while(durum){
            try{
                if(no==temp.no){
                    if(temp.kok.sag!=null){//hata vermemesi icin
                        if(temp.kok.sag.no==no){
                            Yaprak baglanacaksag=temp.sag;
                            temp.kok.sag=temp.sol;
                            temp=temp.sol;
                            while(temp.sag!=null){
                                temp=temp.sag;
                            }
                            temp.sag=baglanacaksag;
                            while(temp.kok!=null){
                                temp=temp.kok;
                            }
                            this.root=temp;
                        }
                    }
                    if(temp.kok.sol!=null){//Hata vermemesi icin
                        if(temp.kok.sol.no==no){
                            Yaprak baglanacaksol=temp.sol;
                            temp.kok.sol=temp.sag;
                            temp=temp.sag;
                            while(temp.sol!=null){
                                temp=temp.sol;
                            }
                            temp.sol=baglanacaksol;
                            while(temp.kok!=null){
                                temp=temp.kok;
                            }
                            this.root=temp;

                        }}
                    bulunma=true;
                    durum=false;
                }
                else if(no>temp.no){
                    temp=temp.sag;
                }
                else if(no<temp.no){
                    temp=temp.sol;
                }
            }
            catch(Exception e){
                durum=false;
            }
            
            
        }
        
        if(bulunma==true){
            
            System.out.println("Silindi");
            }
        else{
            System.out.println("Bulunamadigindan Silinemedi");
        }
    }
    
    
   void preorder(Yaprak dugum){
       if(dugum!=null){
           System.out.print(dugum.no+"|");
           preorder(dugum.sol);
           preorder(dugum.sag);
       }
   }
   
   void inorder(Yaprak dugum){
       if(dugum!=null){
           inorder(dugum.sol);
           System.out.print(dugum.no+"|");
           inorder(dugum.sag);
       }
   }
   
   void postorder(Yaprak dugum){
       if(dugum!=null){
           postorder(dugum.sol);
           postorder(dugum.sag);
           System.out.print(dugum.no+"|");
           
       }
   }
