;Gungor Hoca Kodlama Sablonu

list		p=16f877A
#include	<p16f877A.inc>
	
__CONFIG H'3F31'

;******************************************************************************
;Degiskenler

GECIKME1	EQU	0x20    ;Gecikme 1. Dongu Degiskeni
GECIKME2	EQU	0x21    ;Gecikme 2. Dongu Degiskeni
GECIKME3	EQU	0x22    ;Gecikme 3. Dongu Degiskeni

;******************************************************************************
		
ORG     0x000             	; Baslama Vektoru
nop
goto    BASLA              	; BASLA Etiketine Git

;******************************************************************************
;Gecikme Alt Programi	

GECIKME				; (herbir komut 4mhz de 1mikrosn) = 1 sn GEC?KME (yakla??k)
	MOVLW	0x08
	MOVWF	GECIKME1	
	MOVLW	0x2F
	MOVWF	GECIKME2
	MOVLW	0x01
	MOVWF	GECIKME3
DONGU1
	DECFSZ	GECIKME1,1
	GOTO	$+2
	DECFSZ	GECIKME2,1
	GOTO	$+2
	DECFSZ	GECIKME3,1
	GOTO	DONGU1
	NOP
	NOP
	NOP	
	RETURN
	
;****************************************************************************************************************
;Arttirma ve azaltma alt programlari

AARTTIR
	INCF	PORTA,1		    	; PORTa yi 1 arttir
	return

AAZALT
	DECF	PORTA,1		    	; PORTa yi 1 arttir
	return

;******************************************************************************
;Baslangic

BASLA
	;en agirliksiz set edildi 01.banka gecildi
	BSF	STATUS,RP0
	
	;varsayilan adc pasifleme
	MOVLW	0X06	
	MOVWF	ADCON1
	
	; port a'nin giris cikislarini ayarlama
	MOVLW	b'00110000'
	MOVWF	TRISA		    
	
	; port b'nin giris cikislarini ayarlama
	MOVLW	0X00	
	MOVWF	TRISB		    
	
	;en agirliksiz clear edildi portlar icin 00.banka gecildi
	BCF	STATUS,RP0	
	
	;port a ve b icini temizleme
	CLRF	PORTA
	CLRF	PORTB
	
	;pull-up oldugundan a port cikislara 0, girislere 1 yukle
	MOVLW	b'00110000'	    ; PORT A 'YI BASTA 00000000 YAP
	MOVWF	PORTA
	
	;pull-up oldugundan cikislara 0, girislere 1 yukle
	MOVLW	0X00		    ; PORT B 'YI BASTA 00000000 YAP
	MOVWF	PORTB
	
    DON ;baslangicta butonlari dinleme
	BTFSS	PORTA,4		
	GOTO	ILERISAYICI
	BTFSS	PORTA,5		
	GOTO	GERISAYICI
	GOTO	DON

;******************************************************************************
;Sayicilar
	
ILERISAYICI ;ileri sayma islemi

	CALL	GECIKME		    
	
	; Tüm bitler 1 ise porta'yi arttiracak
	BTFSS   PORTB,7
	GOTO	$+2
	BTFSS   PORTB,6
	GOTO	$+2
	BTFSS   PORTB,5
	GOTO	$+2
	BTFSS   PORTB,4
	GOTO	$+2
	BTFSS   PORTB,3
	GOTO	$+2
	BTFSS   PORTB,2
	GOTO	$+2
	BTFSS   PORTB,1
	GOTO	$+2
	BTFSS   PORTB,0
	GOTO	$+2
	CALL	AARTTIR
	
	; PORTB yi 1 arttir
	INCF	PORTB,1		    	
	
	; butonu dinleyerek pull-up geri sayiciya gecme
	BTFSS	PORTA,5		
	GOTO	GERISAYICI
	GOTO	ILERISAYICI

GERISAYICI ;geri sayma islemi

	CALL	GECIKME
	
	; Tüm bitler 0 ise porta'yi azaltacak
	BTFSC   PORTB,7
	GOTO	$+2
	BTFSC   PORTB,6
	GOTO	$+2
	BTFSC   PORTB,5
	GOTO	$+2
	BTFSC   PORTB,4
	GOTO	$+2
	BTFSC   PORTB,3
	GOTO	$+2
	BTFSC   PORTB,2
	GOTO	$+2
	BTFSC   PORTB,1
	GOTO	$+2
	BTFSC   PORTB,0
	GOTO	$+2
	CALL	AAZALT
	
	;PORTB'yi 1 azalt
	DECF	PORTB,1		    	
	
	; butonu dinleyerek pull-up ileri sayiciya gecme
	BTFSS	PORTA,4		
	GOTO	ILERISAYICI
	GOTO	GERISAYICI
	
END			    ; Program sonu
